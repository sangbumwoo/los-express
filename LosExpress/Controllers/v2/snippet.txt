namespace CommuteService.Utils
{
    public static class QueryStringUtility
    {
        public static string ConvertToQueryString(this object obj)
        {
            string result = string.Empty;

            if (obj != null)
            {
                var properties = from p in obj.GetType().GetProperties()
                                 where p.GetValue(obj, null) != null
                                 select p.GetCustomAttribute<JsonPropertyAttribute>().PropertyName + "=" + HttpUtility.UrlEncode(p.GetValue(obj, null).ToString());

                result = String.Join("&", properties.ToArray());
            }

            return result;
        }
    }
}
